# Deploy JPA server
docker pull maven:3.6.3-jdk-11-slim
docker pull tomcat:9.0.38-jdk11-openjdk-slim-buster

mvn clean install -DskipTests

docker build --rm --build-arg IMAGE_BUILD_TIMESTAMP="%date% %time%" -t pegacorn-buildingblocks-fhirdm:1.0.0-snapshot .

\helm\helm upgrade pegacorn-buildingblocks-fhirdm-site-a --install --namespace site-a --set serviceName=pegacorn-buildingblocks-fhirdm,basePort=32210,imageTag=1.0.0-snapshot,numOfPods=1 helm

Add the following to your hosts file (with Notepad [Adminstrator mode]) - C:\Windows\System32\drivers\etc
127.0.0.1 pegacorn-buildingblocks-fhirdm.site-a

URL: http://pegacorn-buildingblocks-fhirdm.site-a:32210
